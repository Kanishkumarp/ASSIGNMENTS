```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import TimeoutException, NoSuchElementException
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
from typing import Optional

class LearnNestedFrame:
    def __init__(self):
        self.driver: Optional[webdriver.Chrome] = None

    def launch_browser(self) -> None:
        try:
            service = Service(ChromeDriverManager().install())
            self.driver = webdriver.Chrome(service=service)
        except Exception as e:
            print(f"Failed to launch browser: {e}")

    def load_url(self, url: str) -> None:
        if self.driver:
            self.driver.get(url)
            self.driver.maximize_window()
            self.driver.implicitly_wait(15)

    def switch_to_frame(self, frame: str or int) -> None:
        if self.driver:
            try:
                if isinstance(frame, str):
                    element = self.driver.find_element(By.XPATH, frame)
                    self.driver.switch_to.frame(element)
                else:
                    self.driver.switch_to.frame(frame)
            except Exception as e:
                print(f"Failed to switch to frame: {e}")

    def click_button(self, button_id: str) -> None:
        if self.driver:
            try:
                self.driver.find_element(By.ID, button_id).click()
            except Exception as e:
                print(f"Failed to click button: {e}")

    def switch_to_default_content(self) -> None:
        if self.driver:
            self.driver.switch_to.default_content()

    def switch_to_parent_frame(self) -> None:
        if self.driver:
            self.driver.switch_to.parent_frame()

    def execute(self) -> None:
        url = "https://leafground.com/frame.xhtml"
        self.launch_browser()
        self.load_url(url)

        # Switch inside the outer frame
        xpath = "//h5[text()=' Click Me (Inside Nested frame)']/following-sibling::iframe"
        self.switch_to_frame(xpath)

        # Switch inside the inner frame
        self.switch_to_frame(0)

        # Click the button
        self.click_button("Click")

        # Switch to the main page
        self.switch_to_default_content()

        # Switch back to the immediate parent frame
        self.switch_to_parent_frame()

        # Keep the browser open for 10 seconds
        import time
        time.sleep(10)

        # Close the browser
        if self.driver:
            self.driver.quit()


if __name__ == "__main__":
    learn_nested_frame = LearnNestedFrame()
    learn_nested_frame.execute()
```
